% Document formatting for ELIAC Handbook
% Swiss-inspired design with colors, headers, sections, and page layout

% Page geometry configuration - Swiss asymmetric layout
\geometry{
    a4paper,
    left=35mm,      % Larger left margin (Swiss asymmetry)
    right=20mm,     % Smaller right margin
    top=25mm,
    bottom=30mm
}

% Professional color scheme
\definecolor{eliacblue}{RGB}{0,83,161}
\definecolor{eliacgray}{RGB}{102,102,102}
\definecolor{eliacred}{RGB}{204,0,0}
\definecolor{swissred}{RGB}{255,0,0}  % Optional Swiss red accent

% Font setup - Swiss style (serif body, sans headers)
\usepackage{libertine}  % Or substitute with sabon package
\usepackage{helvet}
\usepackage[T1]{fontenc}

% Headers and footers configuration
\pagestyle{fancy}
\fancyhf{}
\fancyhead[L]{\sffamily\textbf{ELIAC Competition Handbook}}
\fancyhead[R]{\sffamily Version 1.0.0}
\fancyfoot[C]{\sffamily\thepage~/~\pageref{LastPage}}
\fancyfoot[R]{\sffamily 2025.09.09}
\renewcommand{\headrulewidth}{0.5pt}
\renewcommand{\footrulewidth}{0.5pt}

% Swiss grid system
\newlength{\gridunit}
\setlength{\gridunit}{4.5mm}  % Swiss standard grid unit
\newcommand{\gridspace}[1]{\vspace{#1\gridunit}}

% Enhanced spacing
\setlength{\parskip}{0.8em}
\setlength{\parindent}{0pt}  % No indent (Swiss style)

% Section formatting - Sans-serif headers with enhanced spacing
\titleformat{\section}[block]
{\sffamily\Large\bfseries\color{eliacblue}}
{\thesection}{1em}{}
\titlespacing*{\section}{0pt}{3.5ex plus 1ex minus .2ex}{2.3ex plus .2ex}

% Subsection formatting - hang style for inline number and title
\titleformat{\subsection}[hang]
{\sffamily\large\bfseries}
{\thesubsection}{1em}{}
\titlespacing*{\subsection}{0pt}{3.25ex plus 1ex minus .2ex}{1.5ex plus .2ex}

% Subsubsection formatting - hang style for better readability
\titleformat{\subsubsection}[hang]
{\sffamily\normalsize\bfseries}
{\thesubsubsection}{1em}{}
\titlespacing*{\subsubsection}{0pt}{2.5ex plus 1ex minus .2ex}{0.5ex plus .2ex}

\setcounter{tocdepth}{2}

% Custom numbering system with letter prefixes
\renewcommand{\thesection}{\arabic{section}}
\renewcommand{\thesubsection}{\thesection.\arabic{subsection}}
\renewcommand{\thesubsubsection}{\thesubsection.\arabic{subsubsection}}

% Fix table of contents spacing for custom numbering
\usepackage{tocloft}
\setlength{\cftsecnumwidth}{2.5em}
\setlength{\cftsubsecnumwidth}{3.5em}
\setlength{\cftsubsubsecnumwidth}{4.5em}

% Page breaks for sections and subsections
\usepackage{titlesec}
\newcommand{\sectionbreak}{\clearpage}

% Enhanced table styling
\usepackage{booktabs}
\usepackage{array}
\renewcommand{\arraystretch}{1.3}  % More breathing room in tables

% Swiss-style table environment
\newenvironment{swisstable}[1]
    {\begin{tabular}{@{}#1@{}}\toprule}
    {\bottomrule\end{tabular}}
